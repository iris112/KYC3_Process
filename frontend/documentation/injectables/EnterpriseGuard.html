import { Injectable } from '@angular/core';
import { Router, ActivatedRouteSnapshot, RouterStateSnapshot, CanActivate } from '@angular/router';
import { AlertService, CallsService} from '../_services/index';

import { Observable } from 'rxjs/Observable';
import { BehaviorSubject } from '../../../node_modules/rxjs';

/** 
 * Implements guard preventing unauthorized user access to the admin pages
*/
@Injectable()
export class EnterpriseGuard implements CanActivate {

  response: any = {};
  private admin = new BehaviorSubject<boolean>(false);

  constructor(private router: Router, private callsService: CallsService, private alertService: AlertService) { }

  get isAdmin() {
    return this.admin.asObservable();
  }

  canActivate(routeSnapshot: ActivatedRouteSnapshot): Observable<boolean> {
    if (localStorage.hasOwnProperty('auth_token') && localStorage.getItem('auth_token').length > 0) {
      return this.callsService.getAdmins().map(
        data => {
          this.response = data;
              //console.log(this.response);
              var isInArray = (this.response.indexOf(localStorage.getItem('username')) > -1);
              if(isInArray == true){
                //console.log("Enterprise guard returned true");
                this.admin.next(true);
                return true;
              }
              else {
                window.alert('You are not authorized to access this URL.');
                this.router.navigate([routeSnapshot.data['enterpriseGuardRedirect']]);
                //console.log("Enterprise guard returned false");
                this.admin.next(false);
                return false;
              }
        },
        error => {
          window.alert('You are not authorized to access this URL.');
          this.router.navigate([routeSnapshot.data['enterpriseGuardRedirect']]);
          //console.log("Enterprise guard returned false");
          this.admin.next(false);
          return false;
        }
      );
    }
  }

}

import { Component, OnInit } from '@angular/core';

import { AlertService } from '../_services/index';

/** 
 * Directive for the alert message
*/
@Component({
    moduleId: module.id,
    selector: 'alert',
    templateUrl: 'alert.component.html'
})
export class AlertComponent {
    message: any;

    constructor(private alertService: AlertService) { }

    ngOnInit() {
        this.alertService.getMessage().subscribe(message => { this.message = message; });
    }
}

<div *ngIf="message" [ngClass]="{ 'alert': message, 'alert-success': message.type === 'success', 'alert-danger': message.type === 'error' }" role="alert">
        <div class="alert-dismissible">
                <button type="button" class="close" (click)="message = null" aria-label="Close">
                                <span aria-hidden="true">&times;</span>
                        </button>
                <span [innerHTML]="message.text"></span>
        </div>
</div>
